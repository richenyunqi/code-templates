# 线段树

线段树是算法竞赛中常用的用来维护区间信息的数据结构。线段树可以在$O(logn)$的时间复杂度内实现单点修改、区间修改、区间查询（区间求和，求区间最大值，求区间最小值）等操作。线段树维护的信息，需要满足可加性，即能以可以接受的速度合并信息和修改信息，包括在使用懒惰标记时，标记也要满足可加性（例如取模就不满足可加性，两个取模操作就不能合并在一起做）。下面的代码以区间求和操作为例。注意，线段树和输入序列下标均需从 1 开始。

## 单点修改，区间查询

```cpp
class SegmentTree {
  public:
    SegmentTree(gg len) : n(len), vi(len + 5), sumTree((len + 5) * 4) {
        for (gg i = 1; i <= n; ++i) {
            cin >> vi[i];  //读取原始输入序列
        }
        init(1, 1, n);  //根据输入序列vi建立线段树
    }
    //将vi[p]增加v
    void realUpdate(gg p, gg v) { update(1, 1, n, p, v); }
    //查询vi的[left,right]区间的和
    gg realGetSum(gg left, gg right) { return getSum(1, 1, n, left, right); }

  private:
    vector<gg> sumTree;  // sumTree存储线段树
    vector<gg> vi;  //原始输入序列
    gg n;  //记录输入序列的长度
    void init(gg root, gg left, gg right) {
        if (left == right) {
            sumTree[root] = vi[left];
            return;
        }
        gg mid = (left + right) / 2;
        init(root * 2, left, mid);
        init(root * 2 + 1, mid + 1, right);
        sumTree[root] = sumTree[root * 2] + sumTree[root * 2 + 1];
    }
    //将vi[p]增加v，[l,r]表示当前root结点包含的区间
    void update(gg root, gg l, gg r, gg p, gg v) {
        if (l == r) {
            sumTree[root] += v;
            return;
        }
        gg mid = (l + r) / 2;
        if (p <= mid) {
            update(root * 2, l, mid, p, v);
        } else {
            update(root * 2 + 1, mid + 1, r, p, v);
        }
        sumTree[root] = sumTree[root * 2] + sumTree[root * 2 + 1];
    }
    //查询vi的[left,right]区间的和
    //[l,r]表示当前root结点包含的区间
    gg getSum(gg root, gg l, gg r, gg left, gg right) {
        if (left <= l and r <= right) {
            return sumTree[root];
        }
        gg mid = (l + r) / 2, ans = 0;
        if (left <= mid) {
            ans += getSum(root * 2, l, mid, left, right);
        }
        if (right > mid) {
            ans += getSum(root * 2 + 1, mid + 1, r, left, right);
        }
        return ans;
    }
};
```

## 区间修改（只有加法），区间查询

```cpp
class SegmentTree {
  public:
    SegmentTree(gg len) : n(len), vi(len + 5), sumTree((len + 5) * 4), lazy((len + 5) * 4) {
        for (gg i = 1; i <= n; ++i) {
            cin >> vi[i];  //读取原始输入序列
        }
        init(1, 1, n);  //根据输入序列vi建立线段树
    }
    //将vi的[left,right]区间的值都加上v
    void realUpdate(gg left, gg right, gg v) {
        update(1, 1, n, left, right, v);
    }
    //查询vi的[left,right]区间的和
    gg realGetSum(gg left, gg right) { return getSum(1, 1, n, left, right); }

  private:
    // sumTree存储线段树，lazy是懒惰标记
    vector<gg> sumTree, lazy;
    vector<gg> vi;  //原始输入序列
    gg n;  //记录输入序列的长度
    //根据输入序列vi建立线段树
    void init(gg root, gg left, gg right) {
        if (left == right) {
            sumTree[root] = vi[left];
            return;
        }
        gg mid = (left + right) / 2;
        init(root * 2, left, mid);
        init(root * 2 + 1, mid + 1, right);
        sumTree[root] = sumTree[root * 2] + sumTree[root * 2 + 1];
    }
    //下传懒惰标记
    void pushdown(gg root, gg l, gg r) {
        gg m = (l + r) / 2;
        if (lazy[root] != 0 and l != r) {
            sumTree[root * 2] += (m - l + 1) * lazy[root],
                lazy[root * 2] += lazy[root];
            sumTree[root * 2 + 1] += (r - m) * lazy[root],
                lazy[root * 2 + 1] += lazy[root];
            lazy[root] = 0;
        }
    }
    //将vi的[left,right]区间的值都加上v
    //[l,r]表示当前root结点包含的区间
    void update(gg root, gg l, gg r, gg left, gg right, gg v) {
        if (left <= l and r <= right) {
            sumTree[root] += (r - l + 1) * v;
            lazy[root] += v;
            return;
        }
        gg mid = (l + r) / 2;
        pushdown(root, l, r);
        if (left <= mid) {
            update(root * 2, l, mid, left, right, v);
        }
        if (right > mid) {
            update(root * 2 + 1, mid + 1, r, left, right, v);
        }
        sumTree[root] = sumTree[root * 2] + sumTree[root * 2 + 1];
    }
    //查询vi的[left,right]区间的和
    //[l,r]表示当前root结点包含的区间
    gg getSum(gg root, gg l, gg r, gg left, gg right) {
        if (left <= l and r <= right) {
            return sumTree[root];
        }
        gg mid = (l + r) / 2, ans = 0;
        pushdown(root, l, r);
        if (left <= mid) {
            ans += getSum(root * 2, l, mid, left, right);
        }
        if (right > mid) {
            ans += getSum(root * 2 + 1, mid + 1, r, left, right);
        }
        return ans;
    }
};
```

## 区间修改（加法和乘法），区间查询

```cpp
class SegmentTree {
  public:
    SegmentTree(gg len, gg p = (1ull << 63) - 1) :
        n(len), sumTree((len + 5) * 4), mul((len + 5) * 4, 1), add((len + 5) * 4), vi(len + 5), mod(p) {
        for (gg i = 1; i <= n; ++i) {
            cin >> vi[i];  //读取原始输入序列
        }
        init(1, 1, n);
    }
    //将vi的[left,right]区间的值都加上v
    void realAdd(gg left, gg right, gg v) {
        update(1, 1, n, left, right, 1, v);
    }
    //将vi的[left,right]区间的值都乘上v
    void realMul(gg left, gg right, gg v) {
        update(1, 1, n, left, right, v, 0);
    }
    //查询vi的[left,right]区间的和
    gg realGetSum(gg left, gg right) { return getSum(1, 1, n, left, right); }

  private:
    gg mod;
    vector<gg> sumTree, mul, add, vi;
    gg n;  //记录输入序列的长度
    //根据输入序列vi建立线段树
    void init(gg root, gg left, gg right) {
        mul[root] = 1, add[root] = 0;
        if (left == right) {
            sumTree[root] = vi[left] % mod;
            return;
        }
        gg mid = (left + right) / 2;
        init(root * 2, left, mid);
        init(root * 2 + 1, mid + 1, right);
        sumTree[root] = (sumTree[root * 2] + sumTree[root * 2 + 1]) % mod;
    }
    //下传懒惰标记
    void pushdown(gg root, gg l, gg r) {
        gg m = (l + r) / 2;
        if ((mul[root] != 1 or add[root] != 0) and l != r) {
            sumTree[root * 2] = (sumTree[root * 2] * mul[root] % mod +
                                 (m - l + 1) * add[root] % mod) %
                                mod;
            sumTree[root * 2 + 1] = (sumTree[root * 2 + 1] * mul[root] % mod +
                                     ((r - m) * add[root]) % mod) %
                                    mod;
            mul[root * 2] = mul[root * 2] * mul[root] % mod;
            mul[root * 2 + 1] = (mul[root * 2 + 1] * mul[root]) % mod;
            add[root * 2] = (add[root * 2] * mul[root] + add[root]) % mod;
            add[root * 2 + 1] =
                (add[root * 2 + 1] * mul[root] + add[root]) % mod;
            mul[root] = 1, add[root] = 0;
        }
    }
    //将vi的[left,right]区间的值都*vmul+vadd
    //[l,r]表示当前root结点包含的区间
    void update(gg root, gg l, gg r, gg left, gg right, gg vmul, gg vadd) {
        if (left <= l and r <= right) {
            sumTree[root] =
                (sumTree[root] * vmul % mod + (r - l + 1) * vadd % mod) % mod;
            mul[root] = mul[root] * vmul % mod;
            add[root] = (add[root] * vmul + vadd) % mod;
            return;
        }
        gg mid = (l + r) / 2;
        pushdown(root, l, r);
        if (left <= mid) {
            update(root * 2, l, mid, left, right, vmul, vadd);
        }
        if (right > mid) {
            update(root * 2 + 1, mid + 1, r, left, right, vmul, vadd);
        }
        sumTree[root] = (sumTree[root * 2] + sumTree[root * 2 + 1]) % mod;
    }
    //查询vi的[left,right]区间的和
    //[l,r]表示当前root结点包含的区间
    gg getSum(gg root, gg l, gg r, gg left, gg right) {
        if (left <= l and r <= right) {
            return sumTree[root];
        }
        gg mid = (l + r) / 2, ans = 0;
        pushdown(root, l, r);
        if (left <= mid) {
            ans += getSum(root * 2, l, mid, left, right);
        }
        if (right > mid) {
            ans += getSum(root * 2 + 1, mid + 1, r, left, right);
        }
        return ans % mod;
    }
};
```
